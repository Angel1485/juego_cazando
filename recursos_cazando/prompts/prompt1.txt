    Sistema de puntaje: • Cada vez que el gato coma, gana 1 punto. • Incrementar el puntaje y mostrarlo en pantalla. • Utilizar funciones de utilitarios.js para actualizar la vista.

function generarAleatorio(min,max){
    let random=Math.random();
    let numero=random*(max-min+1);
    let numeroEntero = Math.ceil(numero);
    numeroEntero = numeroEntero+min-1;
    return numeroEntero
}
function mostrarEnSpan(idSpan,valor){
